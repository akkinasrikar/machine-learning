import numpy as np
import pandas as pd
from scipy import stats
import matplotlib.pyplot as plt
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LinearRegression

dataset=pd.read_csv('house.csv')
y1=dataset.iloc[:,2:3].values
x1=dataset.iloc[:,5:6].values
x=np.array(x1)
y=np.array(y1)
xset=[]
yset=[]
for i in range(len(x)):
    xset.append(float(x[i]))
    yset.append(float(y[i]))
slope,intercept,r_value,p_value,std_err=stats.linregress(xset,yset)
x_train,y_train,x_test,y_test=train_test_split(x,y,test_size=6,random_state=0)
slr=LinearRegression()
slr.fit(x,y)
y_pred=slr.predict(x)
plt.xlabel('sq_feet')
plt.ylabel('price')
xsq=float(input("enter sq feet value>>> "))
y_sam=slope*xsq+intercept-std_err
print(f"predicted house price is {y_sam}")

plt.scatter(x,y,color='red')
plt.plot(x,y_pred)
plt.plot(xsq,y_sam,'g^')
plt.show
